014package org.microbean.interceptor;
015
016import java.lang.invoke.MethodHandle;
017
018import java.lang.reflect.Method;
019
020import java.util.function.Supplier;
021
022import jakarta.interceptor.InvocationContext;
023
024import static java.lang.invoke.MethodHandles.lookup;
025import static java.lang.invoke.MethodHandles.privateLookupIn;
026
027/**
028 * A representation of a Jakarta Interceptors <em>interceptor method</em>.
029 *
030 * @author <a href="https://about.me/lairdnelson/" target="_top">Laird Nelson</a>
031 */
032@FunctionalInterface
033public interface InterceptorMethod {
034
035
036  /**
037   * Performs interception and returns any result.
038   *
039   * @param ic an {@link InvocationContext}; must not be {@code null}
040   *
041   * @return the result of interception, often computed by invoking the {@link InvocationContext#proceed()} method; the
042   * return value may be {@code null}
043   *
044   * @exception Exception if any error occurs
045   */
046  public Object intercept(final InvocationContext ic) throws Exception;
047
048
049  /*
050   * Static methods.
051   */
052
053
054  /**
055   * Returns a new {@link InterceptorMethod} that adapts the supplied {@code static} {@link Method}.
056   *
057   * @param staticMethod a {@code static} {@link Method}; must not be {@code null}
058   *
059   * @return a new {@link InterceptorMethod}; never {@code null}
060   *
061   * @exception NullPointerException if {@code staticMethod} is {@code null}
062   */
063  public static InterceptorMethod of(final Method staticMethod) {
064    return of(staticMethod, null);
065  }
066
067  /**
068   * Returns a new {@link InterceptorMethod} that adapts the supplied {@link Method} and the supplied {@link Supplier}
069   * of its receiver.
070   *
071   * @param m a {@link Method}; must not be {@code null}
072   *
073   * @param targetSupplier a {@link Supplier} of the supplied {@link Method}'s receiver; may be {@code null} in which
074   * case the supplied {@link Method} must be {@code static}
075   *
076   * @return a new {@link InterceptorMethod}; never {@code null}
077   *
078   * @exception NullPointerException if {@code m} is {@code null}
079   *
080   * @exception IllegalStateException if {@linkplain java.lang.invoke.MethodHandles.Lookup#unreflect(Method)
081   * unreflecting} fails
082   */
083  public static InterceptorMethod of(final Method m, final Supplier<?> targetSupplier) {
084    try {
085      return of(privateLookupIn(m.getDeclaringClass(), lookup()).unreflect(m), targetSupplier);
086    } catch (final IllegalAccessException e) {
087      throw new IllegalStateException(e.getMessage(), e);
088    }
089  }
090
091  /**
092   * Returns a new {@link InterceptorMethod} that adapts the supplied {@link MethodHandle}.
093   *
094   * @param receiverlessOrBoundMethodHandle a {@link MethodHandle}; must not be {@code null}; must either not require a
095   * receiver or must be already {@linkplain MethodHandle#bindTo(Object) bound} to one
096   *
097   * @return a new {@link InterceptorMethod}; never {@code null}
098   *
099   * @exception NullPointerException if {@code receiverlessOrBoundMethodHandle} is {@code null}
100   */
101  public static InterceptorMethod of(final MethodHandle receiverlessOrBoundMethodHandle) {
102    return of(receiverlessOrBoundMethodHandle, null);
103  }
104
105  /**
106   * Returns a new {@link InterceptorMethod} that adapts the supplied {@link MethodHandle} and the supplied {@link
107   * Supplier of its receiver}.
108   *
109   * @param mh a {@link MethodHandle}; must not be {@code null}
110   *
111   * @param receiverSupplier a {@link Supplier} of the supplied {@link MethodHandle}'s receiver; may be {@code null} in
112   * which case the supplied {@link MethodHandle} must either not require a receiver or must be already {@linkplain
113   * MethodHandle#bindTo(Object) bound} to one
114   *
115   * @return a new {@link InterceptorMethod}; never {@code null}
116   *
117   * @exception NullPointerException if {@code m} is {@code null}
118   */
119  public static InterceptorMethod of(final MethodHandle mh, final Supplier<?> receiverSupplier) {
120    final Object returnType = mh.type().returnType();
121    if (returnType == void.class || returnType == Void.class) {
122      if (receiverSupplier == null) {
123        return ic -> invokeExact(mh, ic);
124      }
125      final MethodHandle unboundInterceptorMethod = mh.asType(mh.type().changeParameterType(0, Object.class));
126      return ic -> {
127        try {
128          unboundInterceptorMethod.invokeExact(receiverSupplier.get(), ic);
129        } catch (final RuntimeException | Error e) {
130          throw e;
131        } catch (final InterruptedException e) {
132          Thread.currentThread().interrupt();
133          throw new IllegalStateException(e.getMessage(), e);
134        } catch (final Throwable e) {
135          throw new IllegalStateException(e.getMessage(), e);
136        }
137        return null;
138      };
139    } else if (receiverSupplier == null) {
140      return ic -> invokeExact(mh, ic);
141    }
142    final MethodHandle unboundInterceptorMethod = mh.asType(mh.type().changeParameterType(0, Object.class));
143    return ic -> invokeExact(unboundInterceptorMethod, receiverSupplier, ic);
144  }
145
146  private static Object invokeExact(final MethodHandle mh, final InvocationContext ic) {
147    try {
148      return mh.invokeExact(ic);
149    } catch (final RuntimeException | Error e) {
150      throw e;
151    } catch (final InterruptedException e) {
152      Thread.currentThread().interrupt();
153      throw new IllegalStateException(e.getMessage(), e);
154    } catch (final Throwable e) {
155      throw new IllegalStateException(e.getMessage(), e);
156    }
157  }
158
159  private static Object invokeExact(final MethodHandle mh, final Supplier<?> receiverSupplier, final InvocationContext ic) {
160    try {
161      return mh.invokeExact(receiverSupplier.get(), ic);
162    } catch (final RuntimeException | Error e) {
163      throw e;
164    } catch (final InterruptedException e) {
165      Thread.currentThread().interrupt();
166      throw new IllegalStateException(e.getMessage(), e);
167    } catch (final Throwable e) {
168      throw new IllegalStateException(e.getMessage(), e);
169    }
170  }
171
172}
