357  /*
358   * Kitchen sink. Everything is nullable with null meaning, in general, "use a sensible default".
359   */
360
361  private Interception(final Collection<? extends InterceptorMethod> interceptorMethods,
362                       final Function<? super Object[], ?> terminalFunction, // null means lifecycle event
363                       final boolean setTarget, // ignored if interceptorMethods is null or empty
364                       final Supplier<? extends Constructor<?>> constructorBootstrap, // better be memoized
365                       final Supplier<? extends Method> methodBootstrap, // better be memoized
366                       final Supplier<?> targetBootstrap, // ignored if terminalFunction is null or setTarget is true
367                       final Supplier<? extends Object[]> argumentsBootstrap, // ignored if terminalFunction is null
368                       final Consumer<? super Object[]> argumentsValidator, // ignored if terminalFunction is null
369                       final Supplier<?> timerBootstrap, // better be memoized
370                       final Supplier<? extends Set<Annotation>> interceptorBindingsBootstrap) { // better be memoized
371    super();
372    this.data = new ConcurrentHashMap<>();
373    this.interceptorBindingsBootstrap = interceptorBindingsBootstrap == null ? Set::of : interceptorBindingsBootstrap;
374    if (terminalFunction == null) {
375      // Lifecycle event interception. Intercepted by lifecycle callback interceptor methods.
376      //
377      // Examples: @PostConstruct, @PreDestroy
378      if (setTarget) {
379        throw new IllegalArgumentException("terminalFunction: null; setTarget: true");
380      } else if (interceptorMethods == null || interceptorMethods.isEmpty()) {
381        // Pathological (no interception; no terminal function).
382        this.proceeder = Interception::returnNull;
383      } else {
384        final Supplier<?> tb = targetBootstrap == null ? Interception::returnNull : targetBootstrap;
385        final List<InterceptorMethod> ims = List.copyOf(interceptorMethods);
386        this.proceeder = () ->
387          new State(tb)
388            .new Context(ims.iterator())
389            .proceed();
390      }
391      this.constructorBootstrap = Interception::returnNull;
392      this.methodBootstrap = Interception::returnNull;
393      this.timerBootstrap = Interception::returnNull;
394    } else if (setTarget) {
395      // Around-construct interception. Intercepted by lifecycle callback interceptor methods.
396      //
397      // Example: @AroundConstruct
398      if (interceptorMethods == null || interceptorMethods.isEmpty()) {
399        // Pathological (no interception; only the terminal constructor function).
400        final Supplier<? extends Object[]> ab =
401          argumentsBootstrap == null ? Interception::emptyObjectArray : argumentsBootstrap;
402        this.proceeder = () -> terminalFunction.apply(ab.get());
403      } else if (targetBootstrap == null) {
404        final List<InterceptorMethod> ims = List.copyOf(interceptorMethods);
405        this.proceeder = () -> {
406          final State s = new State(argumentsBootstrap, argumentsValidator);
407          final State.Context c = s.new Context(ims.iterator(), args -> s.target(terminalFunction.apply(args)));
408          // This return-value-ignoring ic.proceed() call will look odd to future maintainers (future me).
409          //
410          // An around-construct interceptor method can have one of two signatures:
411          //
412          //   void <METHOD>(InvocationContext)
413          //   Object <METHOD>(InvocationContext) // this one is odd
414          //
415          // The one that returns Object is mainly so you can use the same method to intercept constructors and business
416          // methods. Its return value in a constructor interception scenario is basically undefined. In
417          // around-construct situations I'm not sure why you would rely on this value. We log it here just to make sure
418          // it doesn't get lost.
419          final Object v = c.proceed();
420          final Object t = c.getTarget();
421          if (v != t && LOGGER.isLoggable(DEBUG)) {
422            LOGGER.log(DEBUG, "around-construct proceed() return value: " + v + "; returning getTarget() return value: " + t);
423          }
424          return t;
425        };
426      } else {
427        throw new IllegalArgumentException("setTarget: true; targetBootstrap: " + targetBootstrap);
428      }
429      this.constructorBootstrap = constructorBootstrap == null ? Interception::returnNull : constructorBootstrap;
430      // little extension here for, say, factory methods
431      this.methodBootstrap = methodBootstrap == null ? Interception::returnNull : methodBootstrap;
432      this.timerBootstrap = Interception::returnNull;
433    } else {
434      // Around-invoke or around-timeout interception. Intercepted by business method interceptor methods and timeout
435      // method interceptor methods respectively. You need a target, so you need a targetBootstrap.
436      //
437      // Examples: @AroundInvoke, @AroundTimeout
438      if (interceptorMethods == null || interceptorMethods.isEmpty()) {
439        // Pathological (no interception; only the terminal function).
440        final Supplier<? extends Object[]> ab =
441          argumentsBootstrap == null ? Interception::emptyObjectArray : argumentsBootstrap;
442        this.proceeder = () -> terminalFunction.apply(ab.get());
443      } else {
444        final Supplier<?> tb = targetBootstrap == null ? Interception::returnNull : targetBootstrap;
445        final List<InterceptorMethod> ims = List.copyOf(interceptorMethods);
446        this.proceeder = () ->
447          new State(tb, argumentsBootstrap, argumentsValidator, true)
448            .new Context(ims.iterator(), terminalFunction)
449            .proceed();
450      }
451      this.constructorBootstrap = Interception::returnNull;
452      this.methodBootstrap = methodBootstrap == null ? Interception::returnNull : methodBootstrap;
453      this.timerBootstrap = timerBootstrap == null ? Interception::returnNull : timerBootstrap;
454    }
455  }

546  /**
547   * Creates and returns a {@link Function} encapsulating the supplied {@link MethodHandle}.
548   *
549   * @param mh a {@link MethodHandle}; must not be {@code null}. If {@code receiverSupplier} is non-{@code null}, then
550   * {@code mh} must accept at least one leading argument (a receiver supplied by the supplied {@code
551   * receiverSupplier}).
552   *
553   * @param receiverSupplier a {@link Supplier} of the receiver for the supplied {@link MethodHandle}; may be {@code
554   * null} in which case the supplied {@link MethodHandle} must be receiverless
555   *
556   * @return a {@link Function} encapsulating the supplied {@link MethodHandle}; never {@code null}
557   *
558   * @exception NullPointerException if {@code mh} is {@code null}
559   */
560  public static final Function<Object[], Object> terminalFunctionOf(MethodHandle mh, final Supplier<?> receiverSupplier) {
561    mh = mh.asType(mh.type().changeReturnType(Object.class));
562    MethodType mt = mh.type();
563
564    final int pc = mt.parameterCount();
565
566    final MethodHandle terminalFunction;
567
568    if (receiverSupplier == null) {
569      // Static
570      switch (pc) {
571      case 0:
572        terminalFunction = mh;
573        return ps -> invokeUnchecked(terminalFunction::invokeExact);
574      default:
575        terminalFunction = mh.asSpreader(Object[].class, pc);
576        return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(ps));
577      }
578    }
579
580    // Virtual
581    mh = mh.asType(mt.changeParameterType(0, Object.class));
582    mt = mh.type();
583
584    switch (pc) {
585    case 0:
586      throw new AssertionError(); // changeParameterType() would have blown up
587    case 1:
588      terminalFunction = mh;
589      return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(receiverSupplier.get()));
590    default:
591      terminalFunction = mh.asSpreader(Object[].class, pc - 1);
592      return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(receiverSupplier.get(), ps));
593    }
594  }

596  /**
597   * A convenience method that ensures that every element of the supplied {@code arguments} array can be assigned to a
598   * reference bearing the corresponding {@link Class} drawn from the supplied {@code parameterTypes} array.
599   *
600   * <p>Boxing, unboxing and widening conversions are taken into consideration.</p>
601   *
602   * <p>This method implements the logic implied, but nowhere actually specified, by the contract of {@link
603   * InvocationContext#setParameters(Object[])}.</p>
604   *
605   * @param parameterTypes an array of {@link Class} instances; may be {@code null}; must not contain {@code null}
606   * elements or {@code void.class}; must have a length equal to that of the supplied {@code arguments} array
607   *
608   * @param arguments an array of {@link Object}s; may be {@code null}; must have a length equal to that of the supplied
609   * {@code parameterTypes} array
610   *
611   * @exception IllegalArgumentException if validation fails, i.e. if the length of {@code parameterTypes} is not equal
612   * to the length of {@code arguments}, or if an element of {@code parameterTypes} is {@code null} or {@code
613   * void.class}, or if not every element of the supplied {@code arguments} array can be assigned to a reference bearing
614   * the corresponding {@link Class} drawn from the supplied {@code parameterTypes} array
615   */
616  public static final void validate(final Class<?>[] parameterTypes, final Object[] arguments) {
617    final int parameterTypesLength = parameterTypes == null ? 0 : parameterTypes.length;
618    final int argumentsLength = arguments == null ? 0 : arguments.length;
619    if (argumentsLength != parameterTypesLength) {
620      throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
621                                         "; arguments: " + Arrays.toString(arguments));
622    }
623    for (int i = 0; i < argumentsLength; i++) {
624      final Class<?> parameterType = parameterTypes[i];
625      if (parameterType == null || parameterType == void.class) {
626        throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
627                                           "; arguments: " + Arrays.toString(arguments) +
628                                           "; parameter type: " + parameterType);
629      }
630      final Object argument = arguments[i];
631      if (argument == null) {
632        if (parameterType.isPrimitive() || parameterType != Void.class) {
633          throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
634                                             "; arguments: " + Arrays.toString(arguments) +
635                                             "; parameter type: " + parameterType.getName() +
636                                             "; argument: null");
637        }
638      } else {
639        final Class<?> argumentType = argument.getClass();
640        if (parameterType != argumentType) {
641          if (parameterType == boolean.class) {
642            if (argumentType != Boolean.class) {
643              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
644                                                 "; arguments: " + Arrays.toString(arguments) +
645                                                 "; parameter type: boolean" +
646                                                 "; argument type: " + argumentType.getName());
647            }
648          } else if (parameterType == Boolean.class) {
649            if (argumentType != boolean.class) {
650              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
651                                                 "; arguments: " + Arrays.toString(arguments) +
652                                                 "; parameter type: java.lang.Boolean" +
653                                                 "; argument type: " + argumentType.getName());
654            }
655          } else if (parameterType == byte.class) {
656            if (argumentType != Byte.class) {
657              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
658                                                 "; arguments: " + Arrays.toString(arguments) +
659                                                 "; parameter type: byte" +
660                                                 "; argument type: " + argumentType.getName());
661            }
662          } else if (parameterType == Byte.class) {
663            if (argumentType != byte.class) {
664              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
665                                                 "; arguments: " + Arrays.toString(arguments) +
666                                                 "; parameter type: java.lang.Byte" +
667                                                 "; argument type: " + argumentType.getName());
668            }
669          } else if (parameterType == char.class) {
670            if (argumentType != Character.class) {
671              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
672                                                 "; arguments: " + Arrays.toString(arguments) +
673                                                 "; parameter type: char" +
674                                                 "; argument type: " + argumentType.getName());
675            }
676          } else if (parameterType == Character.class) {
677            if (argumentType != char.class) {
678              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
679                                                 "; arguments: " + Arrays.toString(arguments) +
680                                                 "; parameter type: java.lang.Character" +
681                                                 "; argument type: " + argumentType.getName());
682            }
683          } else if (parameterType == double.class) {
684            if (argumentType != byte.class &&
685                argumentType != char.class &&
686                argumentType != Double.class &&
687                argumentType != float.class &&
688                argumentType != int.class &&
689                argumentType != long.class &&
690                argumentType != short.class) {
691              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
692                                                 "; arguments: " + Arrays.toString(arguments) +
693                                                 "; parameter type: double" +
694                                                 "; argument type: " + argumentType.getName());
695            }
696          } else if (parameterType == Double.class) {
697            if (argumentType != double.class) {
698              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
699                                                 "; arguments: " + Arrays.toString(arguments) +
700                                                 "; parameter type: java.lang.Double" +
701                                                 "; argument type: " + argumentType.getName());
702            }
703          } else if (parameterType == float.class) {
704            if (argumentType != byte.class &&
705                argumentType != char.class &&
706                argumentType != Float.class &&
707                argumentType != int.class &&
708                argumentType != long.class &&
709                argumentType != short.class) {
710              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
711                                                 "; arguments: " + Arrays.toString(arguments) +
712                                                 "; parameter type: float" +
713                                                 "; argument type: " + argumentType.getName());
714            }
715          } else if (parameterType == Float.class) {
716            if (argumentType != float.class) {
717              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
718                                                 "; arguments: " + Arrays.toString(arguments) +
719                                                 "; parameter type: java.lang.Float" +
720                                                 "; argument type: " + argumentType.getName());
721            }
722          } else if (parameterType == int.class) {
723            if (argumentType != byte.class &&
724                argumentType != char.class &&
725                argumentType != Integer.class &&
726                argumentType != short.class) {
727              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
728                                                 "; arguments: " + Arrays.toString(arguments) +
729                                                 "; parameter type: int; argument type: " + argumentType.getName());
730            }
731          } else if (parameterType == Integer.class) {
732            if (argumentType != int.class) {
733              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
734                                                 "; arguments: " + Arrays.toString(arguments) +
735                                                 "; parameter type: java.lang.Integer" +
736                                                 "; argument type: " + argumentType.getName());
737            }
738          } else if (parameterType == long.class) {
739            if (argumentType != byte.class &&
740                argumentType != char.class &&
741                argumentType != int.class &&
742                argumentType != Long.class &&
743                argumentType != short.class) {
744              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
745                                                 "; arguments: " + Arrays.toString(arguments) +
746                                                 "; parameter type: long" +
747                                                 "; argument type: " + argumentType.getName());
748            }
749          } else if (parameterType == Long.class) {
750            if (argumentType != long.class) {
751              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
752                                                 "; arguments: " + Arrays.toString(arguments) +
753                                                 "; parameter type: java.lang.Long" +
754                                                 "; argument type: " + argumentType.getName());
755            }
756          } else if (parameterType == short.class) {
757            if (argumentType != byte.class &&
758                argumentType != Short.class) {
759              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
760                                                 "; arguments: " + Arrays.toString(arguments) +
761                                                 "; parameter type: byte" +
762                                                 "; argument type: " + argumentType.getName());
763            }
764          } else if (parameterType == Short.class) {
765            if (argumentType != short.class) {
766              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
767                                                 "; arguments: " + Arrays.toString(arguments) +
768                                                 "; parameter type: java.lang.Short" +
769                                                 "; argument type: " + argumentType.getName());
770            }
771          } else if (parameterType == Void.class || !parameterType.isAssignableFrom(argumentType)) {
772            throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
773                                               "; arguments: " + Arrays.toString(arguments) +
774                                               "; parameter type: " + parameterType.getName() +
775                                               "; argument type: " + argumentType.getName());
776          }
777        }
778      }
779    }

824  private final class State {
825
826    private volatile Optional<Object[]> arguments;
827
828    private final Supplier<? extends Object[]> argumentsReader;
829
830    private final Consumer<? super Object[]> argumentsInstaller;
831
832    private volatile Optional<Object> target;
833
834    private final Supplier<?> targetReader;
835
836    // "update and get" semantics
837    private final Function<? super Object, ?> targetInstaller;
838
839    private State(final Supplier<?> targetBootstrap) {
840      this(targetBootstrap, null, null, false);
841    }
842
843    private State(final Supplier<? extends Object[]> argumentsBootstrap,
844                  final Consumer<? super Object[]> argumentsValidator) {
845      this(null, argumentsBootstrap, argumentsValidator, true);
846    }
847
848    private State(final Supplier<?> targetBootstrap, // nullable
849                  final Supplier<? extends Object[]> argumentsBootstrap, // ignored if mutableArguments is false
850                  final Consumer<? super Object[]> argumentsValidator, // ignored if mutable arguments is false
851                  final boolean mutableArguments) {
852      super();
853      if (targetBootstrap == null) {
854        this.target = Optional.empty();
855        this.targetInstaller = target -> {
856          this.target = Optional.ofNullable(target); // volatile write
857          return target;
858        };
859        this.targetReader = () -> this.target.orElse(null); // volatile read
860      } else {
861        this.targetInstaller = Interception::throwIllegalStateException;
862        this.targetReader = () -> {
863          Optional<Object> target = this.target; // volatile read
864          if (target == null) {
865            target = Optional.ofNullable(targetBootstrap.get());
866            this.target = target; // volatile write
867          }
868          return target.orElse(null);
869        };
870      }
871      if (mutableArguments) {
872        if (argumentsBootstrap == null) {
873          this.arguments = OPTIONAL_EMPTY_OBJECT_ARRAY;
874          this.argumentsReader = this.arguments::orElseThrow; // volatile read
875        } else {
876          this.argumentsReader = () -> {
877            Optional<Object[]> arguments = this.arguments; // volatile read
878            if (arguments == null) {
879              final Object[] a = argumentsBootstrap.get();
880              arguments = a == null || a.length == 0 ? OPTIONAL_EMPTY_OBJECT_ARRAY : Optional.of(a);
881              this.arguments = arguments; // volatile write
882            }
883            return arguments.orElse(null);
884          };
885        }
886        final Consumer<? super Object[]> av = argumentsValidator == null ? Interception::sink : argumentsValidator;
887        this.argumentsInstaller = args -> {
888          if (args == null || args.length == 0) {
889            av.accept(EMPTY_OBJECT_ARRAY);
890            this.arguments = OPTIONAL_EMPTY_OBJECT_ARRAY; // volatile write
891          } else {
892            av.accept(args);
893            this.arguments = Optional.of(args.clone()); // volatile write
894          }
895        };
896      } else {
897        this.argumentsReader = Interception::throwIllegalStateException;
898        this.argumentsInstaller = Interception::throwIllegalStateException;
899      }
900    }
901
902    private final Object[] arguments() {
903      return this.argumentsReader.get();
904    }
905
906    private final void arguments(final Object[] arguments) {
907      this.argumentsInstaller.accept(arguments);
908    }
909
910    private final Object target() {
911      return this.targetReader.get();
912    }
913
914    // "update and get" semantics
915    private final Object target(final Object target) {
916      return this.targetInstaller.apply(target);
917    }
