309               final Supplier<?> targetSupplier,

311               final boolean setTarget, // is the terminal function effectively a constructor?
312               final Object[] arguments) {
313    this(interceptorMethods,
314         terminalFunction,
315         setTarget,
316         new ConcurrentHashMap<>(),

318         Chain::returnNull, // method supplier

325  private Chain(List<? extends InterceptorMethod> interceptorMethods,
326                final Function<? super Object[], ?> terminalFunction,
327                final boolean setTarget,
328                final ConcurrentMap<String, Object> contextData,
329                final Supplier<? extends Constructor<?>> constructorSupplier,
330                final Supplier<? extends Method> methodSupplier,
331                final Supplier<?> targetSupplier,
332                final Object[] arguments,
333                final Supplier<?> timerSupplier,
334                final AtomicReference<Object> targetReference) {
335    super();
336    this.contextData = contextData == null ? new ConcurrentHashMap<>() : contextData;
337    this.constructorSupplier = constructorSupplier == null ? Chain::returnNull : constructorSupplier;
338    this.methodSupplier = methodSupplier == null ? Chain::returnNull : methodSupplier;
339    this.arguments = arguments == null ? EMPTY_OBJECT_ARRAY : arguments;
340    this.timerSupplier = timerSupplier == null ? Chain::returnNull : timerSupplier;
341    this.targetReference = targetReference == null ? new AtomicReference<>() : targetReference;
342    this.targetSupplier = targetSupplier == null ? Chain::returnNull : targetSupplier;
343    if (interceptorMethods == null || interceptorMethods.isEmpty()) {
344      Objects.requireNonNull(terminalFunction, "terminalFunction");
345      if (setTarget) {
346        this.proceedImplementation = () -> this.targetReference.updateAndGet(v -> terminalFunction.apply(this.getParameters()));
347      } else {
348        this.proceedImplementation = () -> terminalFunction.apply(this.getParameters());
349      }
350    } else {
351      interceptorMethods = List.copyOf(interceptorMethods);
352      final InterceptorMethod im = interceptorMethods.get(0);
353      final int size = interceptorMethods.size();
354      final List<? extends InterceptorMethod> ims = size == 1 ? List.of() : interceptorMethods.subList(1, size);
355      this.proceedImplementation = () -> {
356        try {
357          return im.intercept(new Chain(ims,
358                                        terminalFunction,
359                                        setTarget,
360                                        this.contextData,
361                                        this::getConstructor,
362                                        this::getMethod,
363                                        this.targetSupplier,
364                                        this.arguments,
365                                        this::getTimer,
366                                        this.targetReference));
367        } catch (final RuntimeException | Error e) {
368          throw e;
369        } catch (final InterruptedException e) {
370          Thread.currentThread().interrupt();
371          throw new IllegalStateException(e.getMessage(), e);
372        } catch (final Exception e) {
373          throw new IllegalStateException(e.getMessage(), e);
374        }
375      };
376    }
377  }

421  /**
422   * Returns the target instance, if available, or {@code null}.
423   *
424   * @return the target instance, if available, or {@code null}
425   */
426  @Override
427  public final Object getTarget() {
428    Object target = this.targetReference.get();
429    if (target == null) {
430      target = this.targetSupplier.get();
431      return target == null || this.targetReference.compareAndSet(null, target) ? target : this.targetReference.get();
432    }
433    return target;
434  }

579  /**
580   * Creates and returns a {@link Function} encapsulating the supplied {@link MethodHandle}.
581   *
582   * @param mh a {@link MethodHandle}; must not be {@code null}. If {@code mh} is a {@link MethodHandle} requiring a
583   * receiver, then the supplied {@code receiverSupplier} will be used to supply its receiver
584   *
585   * @param receiverSupplier a {@link Supplier} of the receiver for the supplied {@link MethodHandle}; may be {@code
586   * null} in which case the supplied {@link MethodHandle} must be receiverless
587   *
588   * @return a {@link Function} encapsulating the supplied {@link MethodHandle}; never {@code null}
589   *
590   * @exception NullPointerException if {@code mh} is {@code null}
591   */
592  public static final Function<Object[], Object> terminalFunctionOf(MethodHandle mh, final Supplier<?> receiverSupplier) {
593    mh = mh.asType(mh.type().changeReturnType(Object.class));
594    MethodType mt = mh.type();
595    final int pc = mt.parameterCount();
596
597    final MethodHandle terminalFunction;
598
599    if (receiverSupplier == null) {
600      // Static
601      switch (pc) {
602      case 0:
603        terminalFunction = mh;
604        return ps -> invokeUnchecked(() -> terminalFunction.invokeExact());
605      default:
606        terminalFunction = pc == 1 && mt.parameterType(0) == Object[].class ? mh : mh.asSpreader(Object[].class, pc);
607        return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(ps));
608      }
609    }
610
611    // Virtual
612    mh = mh.asType(mt.changeParameterType(0, Object.class));
613    mt = mh.type();
614
615    switch (pc) {
616    case 1:
617      terminalFunction = mh;
618      return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(receiverSupplier.get()));
619    default:
620      terminalFunction = pc == 2 && mt.parameterType(1) == Object[].class ? mh : mh.asSpreader(Object[].class, pc - 1);
621      return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(receiverSupplier.get(), ps));
622    }
623  }
