432  // Everything is nullable.
433  private Chain(List<? extends InterceptorMethod> interceptorMethods,
434                final Function<? super Object[], ?> terminalFunction,
435                final boolean setTarget,
436                final ConcurrentMap<String, Object> contextData,
437                final Supplier<? extends Constructor<?>> constructorSupplier,
438                final Supplier<? extends Method> methodSupplier,
439                final Supplier<?> targetSupplier,
440                final Supplier<? extends Object[]> argumentsSupplier,
441                final Consumer<? super Object[]> argumentsValidator,
442                final Supplier<?> timerSupplier,
443                final Chain targetHost) {
444    super();
445    this.contextData = contextData == null ? new ConcurrentHashMap<>() : contextData;
446    this.constructorSupplier = constructorSupplier == null ? Chain::returnNull : constructorSupplier;
447    this.methodSupplier = methodSupplier == null ? Chain::returnNull : methodSupplier;
448    this.argumentsSupplier = argumentsSupplier == null ? Chain::emptyObjectArray : argumentsSupplier;
449    this.timerSupplier = timerSupplier == null ? Chain::returnNull : timerSupplier;
450    this.targetSupplier = targetSupplier == null ? Chain::returnNull : targetSupplier;
451    this.targetHost = targetHost == null ? this : targetHost;
452    if (interceptorMethods == null || interceptorMethods.isEmpty()) {
453      Objects.requireNonNull(terminalFunction, "terminalFunction");
454      if (argumentsValidator == null) {
455        this.setParametersImplementation = args -> this.setArguments(Chain::sink, args);
456      } else {
457        this.setParametersImplementation = args -> this.setArguments(argumentsValidator, args);
458      }
459      if (setTarget) {
460        this.proceedImplementation = () -> {
461          final Object t = terminalFunction.apply(this.getParameters());
462          TARGET.setVolatile(this.targetHost, t); // volatile write
463          return t;
464        };
465      } else {
466        this.proceedImplementation = () -> terminalFunction.apply(this.getParameters());
467      }
468    } else {
469      if (terminalFunction == null) {
470        this.setParametersImplementation = Chain::throwIllegalStateException;
471      } else if (argumentsValidator == null) {
472        this.setParametersImplementation = args -> this.setArguments(Chain::sink, args);
473      } else {
474        this.setParametersImplementation = args -> this.setArguments(argumentsValidator, args);
475      }
476      interceptorMethods = List.copyOf(interceptorMethods);
477      final InterceptorMethod im = interceptorMethods.get(0);
478      final int size = interceptorMethods.size();
479      final List<? extends InterceptorMethod> ims = size == 1 ? List.of() : interceptorMethods.subList(1, size);
480      this.proceedImplementation =
481        () -> im.intercept(new Chain(ims,
482                                     terminalFunction,
483                                     setTarget,
484                                     this.contextData,
485                                     this::getConstructor,
486                                     this::getMethod,
487                                     this.targetSupplier,
488                                     this.argumentsSupplier,
489                                     argumentsValidator,
490                                     this::getTimer,
491                                     this.targetHost));
492    }
493  }

535  /**
536   * Returns any arguments in effect for the current interception.
537   *
538   * <p>For historical reasons, the Jakarta Interceptors specification refers, incorrectly, to arguments as
539   * "parameters".</p>
540   *
541   * @return any arguments in effect for the current interception; never {@code null}
542   *
543   * @exception IllegalStateException if invoked within a lifecycle callback method that is not an {@link
544   * jakarta.interceptor.AroundConstruct AroundConstruct} callback
545   *
546   * @see #setParameters(Object[])
547   */
548  @Override
549  public final Object[] getParameters() {
550    // Cloning etc. is not necessary; this whole API is stupid
551    final Object[] arguments = this.arguments; // volatile read
552    if (arguments == null) {
553      try {
554        this.setParameters(this.argumentsSupplier.get());
555      } catch (final IllegalArgumentException e) {
556        throw new IllegalStateException(e.getMessage(), e);
557      }
558      return this.arguments; // volatile read
559    }
560    return arguments;
561  }

563  /**
564   * Returns the target instance, if available, or {@code null}.
565   *
566   * @return the target instance, if available, or {@code null}
567   */
568  @Override
569  public final Object getTarget() {
570    Object target = this.targetHost.target; // volatile read
571    if (target == null) {
572      target = this.targetSupplier.get();
573      if (target != null && TARGET.compareAndSet(this.targetHost, null, target)) { // volatile write
574        target = this.targetHost.target; // volatile read
575      }
576    }
577    return target;
578  }

652  private final void setArguments(final Consumer<? super Object[]> argumentsValidator, final Object[] arguments) {
653    if (arguments == null) {
654      argumentsValidator.accept(EMPTY_OBJECT_ARRAY);
655      this.arguments = EMPTY_OBJECT_ARRAY; // volatile write
656    } else {
657      // Cloning etc. is not necessary; this whole API is stupid
658      argumentsValidator.accept(arguments);
659      this.arguments = arguments; // volatile write
660    }
661  }

705   * @param receiverSupplier a {@link Supplier} of the receiver for the supplied {@link Method}; may be {@code null} in
706   * which case the supplied {@link Method} must be {@code static}
707   *
708   * @return a {@link Function} encapsulating the supplied {@link Method}; never {@code null}
709   *
710   * @exception NullPointerException if {@code m} is {@code null}
711   *
712   * @exception IllegalAccessException if {@linkplain Lookup#unreflect(Method) unreflecting} fails
713   */
714  public static final Function<Object[], Object> terminalFunctionOf(final Method m, final Supplier<?> receiverSupplier)
715    throws IllegalAccessException {

733  /**
734   * Creates and returns a {@link Function} encapsulating the supplied {@link MethodHandle}.
735   *
736   * @param mh a {@link MethodHandle}; must not be {@code null}. If {@code mh} is a {@link MethodHandle} requiring a
737   * receiver, then the supplied {@code receiverSupplier}, if non-{@code null}, will be used to supply its receiver
738   *
739   * @param receiverSupplier a {@link Supplier} of the receiver for the supplied {@link MethodHandle}; may be {@code
740   * null} in which case the supplied {@link MethodHandle} must be receiverless
741   *
742   * @return a {@link Function} encapsulating the supplied {@link MethodHandle}; never {@code null}
743   *
744   * @exception NullPointerException if {@code mh} is {@code null}
745   */
746  public static final Function<Object[], Object> terminalFunctionOf(MethodHandle mh, final Supplier<?> receiverSupplier) {
747    mh = mh.asType(mh.type().changeReturnType(Object.class));
748    MethodType mt = mh.type();
749    final int pc = mt.parameterCount();
750
751    final MethodHandle terminalFunction;
752
753    if (receiverSupplier == null) {
754      // Static
755      switch (pc) {
756      case 0:
757        terminalFunction = mh;
758        return ps -> invokeUnchecked(() -> terminalFunction.invokeExact());
759      default:
760        terminalFunction = mh.asSpreader(Object[].class, pc);
761        return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(ps));
762      }
763    }
764
765    // Virtual
766    mh = mh.asType(mt.changeParameterType(0, Object.class));
767    mt = mh.type();
768
769    switch (pc) {
770    case 1:
771      terminalFunction = mh;
772      return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(receiverSupplier.get()));
773    default:
774      terminalFunction = mh.asSpreader(Object[].class, pc - 1);
775      return ps -> invokeUnchecked(() -> terminalFunction.invokeExact(receiverSupplier.get(), ps));
776    }
777  }

799  /**
800   * A convenience method that ensures that every element of the supplied {@code arguments} array can be assigned to a
801   * reference bearing the corresponding {@link Class} drawn from the supplied {@code parameterTypes} array.
802   *
803   * <p>Boxing, unboxing and widening conversions are taken into consideration.</p>
804   *
805   * <p>This method implements the logic implied, but nowhere actually specified, by the contract of {@link
806   * InvocationContext#setParameters(Object[])}.</p>
807   *
808   * @param parameterTypes an array of {@link Class} instances; may be {@code null}; must not contain {@code null}
809   * elements or {@code void.class}; must have a length equal to that of the supplied {@code arguments} array
810   *
811   * @param arguments an array of {@link Object}s; may be {@code null}; must have a length equal to that of the supplied
812   * {@code parameterTypes} array
813   *
814   * @exception IllegalArgumentException if not every element of the supplied {@code arguments} array can be assigned to
815   * a reference bearing the corresponding {@link Class} drawn from the supplied {@code parameterTypes} array
816   *
817   * @see #setParameters(Object[])
818   */
819  public static final void validate(final Class<?>[] parameterTypes, final Object[] arguments) {
820    final int parameterTypesLength = parameterTypes == null ? 0 : parameterTypes.length;
821    final int argumentsLength = arguments == null ? 0 : arguments.length;
822    if (argumentsLength != parameterTypesLength) {
823      throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
824                                         "; arguments: " + Arrays.toString(arguments));
825    }
826    for (int i = 0; i < argumentsLength; i++) {
827      final Class<?> parameterType = parameterTypes[i];
828      if (parameterType == null || parameterType == void.class) {
829        throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
830                                           "; arguments: " + Arrays.toString(arguments) +
831                                           "; parameter type: " + parameterType);
832      }
833      final Object argument = arguments[i];
834      if (argument == null) {
835        if (parameterType.isPrimitive() || parameterType != Void.class) {
836          throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
837                                             "; arguments: " + Arrays.toString(arguments) +
838                                             "; parameter type: " + parameterType.getName() +
839                                             "; argument: null");
840        }
841      } else {
842        final Class<?> argumentType = argument.getClass();
843        if (parameterType != argumentType) {
844          if (parameterType == boolean.class) {
845            if (argumentType != Boolean.class) {
846              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
847                                                 "; arguments: " + Arrays.toString(arguments) +
848                                                 "; parameter type: boolean" +
849                                                 "; argument type: " + argumentType.getName());
850            }
851          } else if (parameterType == Boolean.class) {
852            if (argumentType != boolean.class) {
853              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
854                                                 "; arguments: " + Arrays.toString(arguments) +
855                                                 "; parameter type: java.lang.Boolean" +
856                                                 "; argument type: " + argumentType.getName());
857            }
858          } else if (parameterType == byte.class) {
859            if (argumentType != Byte.class) {
860              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
861                                                 "; arguments: " + Arrays.toString(arguments) +
862                                                 "; parameter type: byte" +
863                                                 "; argument type: " + argumentType.getName());
864            }
865          } else if (parameterType == Byte.class) {
866            if (argumentType != byte.class) {
867              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
868                                                 "; arguments: " + Arrays.toString(arguments) +
869                                                 "; parameter type: java.lang.Byte" +
870                                                 "; argument type: " + argumentType.getName());
871            }
872          } else if (parameterType == char.class) {
873            if (argumentType != Character.class) {
874              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
875                                                 "; arguments: " + Arrays.toString(arguments) +
876                                                 "; parameter type: char" +
877                                                 "; argument type: " + argumentType.getName());
878            }
879          } else if (parameterType == Character.class) {
880            if (argumentType != char.class) {
881              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
882                                                 "; arguments: " + Arrays.toString(arguments) +
883                                                 "; parameter type: java.lang.Character" +
884                                                 "; argument type: " + argumentType.getName());
885            }
886          } else if (parameterType == double.class) {
887            if (argumentType != byte.class &&
888                argumentType != char.class &&
889                argumentType != Double.class &&
890                argumentType != float.class &&
891                argumentType != int.class &&
892                argumentType != long.class &&
893                argumentType != short.class) {
894              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
895                                                 "; arguments: " + Arrays.toString(arguments) +
896                                                 "; parameter type: double" +
897                                                 "; argument type: " + argumentType.getName());
898            }
899          } else if (parameterType == Double.class) {
900            if (argumentType != double.class) {
901              throw new IllegalArgumentException("parameter types: " + Arrays.toString(parameterTypes) +
902                                                 "; arguments: " + Arrays.toString(arguments) +
903                                                 "; parameter type: java.lang.Double" +
904                                                 "; argument type: " + argumentType.getName());
905            }
906          } else if (parameterType == float.class) {
907            if (argumentType != byte.class &&
908                argumentType != char.class &&
909                argumentType != Float.class &&
910                argumentType != int.class &&
911                argumentType != long.class &&
912                argumentType != short.class) {
